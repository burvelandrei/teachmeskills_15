create table authors
(
    id int primary key,
    first_name varchar(50),
    last_name varchar(50)
);

create table books
(
    id int primary key,
    title varchar(50),
    author_id int,
    publication_year varchar(10),
    foreign key(author_id) references authors(id)
)

create table sales
(
	id int primary key,
	book_id int,
	quantity int,
	foreign key(book_id) references books(id)
);
